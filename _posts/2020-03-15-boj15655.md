# 문제 정보
백준 ( BOJ ) 15655 N과M (5)

문제 풀이 : 수열

문제 출처 : https://www.acmicpc.net/problem/15655


N과 M 문제 종합 보기 : https://mountrivers.github.io/bsequence/


```
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

int n, m;
int base;
vector<int> v(8, 0);
vector<int> result;

void fun(int roopedTime, int startPoint) {
	if (roopedTime == m) {
		for (int i = 0; i < m; i++) {
			printf("%d ", result[v[i]]);
		}
		printf("\n");
	}
	else {
		for (int i = startPoint; i < n; i++) {
			v[roopedTime] = i;
			fun(roopedTime + 1, i + 1);
		}
	}
}

int main() {
	int temp;
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		cin >> temp;
		result.push_back(temp);
	}
	sort(result.begin(), result.end());
	fun(0, 0);


	return 0;
}
```
